    <script>
        // تفاعل تحريك خرزات المعداد الياباني عند النقر عليها لتبدو حية وتفاعلية
        function toggleBead(bead, value) {
            if (bead.style.transform === 'translateY(15px)' || bead.style.transform === 'translateY(-15px)') {
                bead.style.transform = 'translateY(0px)';
            } else {
                bead.style.transform = value > 0 ? 'translateY(15px)' : 'translateY(-15px)';
            }
        }

        // التحقق من إجابة اللغز السريع
        function checkRiddle(button, isCorrect) {
            const feedback = document.getElementById('riddle-feedback');
            
            // إعادة ضبط جميع الأزرار أولاً لتهيئة المظهر
            button.parentElement.querySelectorAll('button').forEach(btn => {
                btn.className = "bg-gray-50 border border-gray-200 py-2 rounded-xl text-sm font-bold transition";
            });

            if (isCorrect) {
                button.className = "bg-green-500 border border-green-500 text-white py-2 rounded-xl text-sm font-bold transition shadow-md shadow-green-500/20";
                feedback.textContent = "🎉 إجابة صحيحة وممتازة! (5 × 5 = 25 + 5 = 30)";
                feedback.className = "text-xs text-center mt-3 font-semibold text-green-600";
            } else {
                button.className = "bg-red-500 border border-red-500 text-white py-2 rounded-xl text-sm font-bold transition shadow-md shadow-red-500/20";
                feedback.textContent = "❌ حاول مرة أخرى، فكّر بها جيداً!";
                feedback.className = "text-xs text-center mt-3 font-semibold text-red-500";
            }
        }
    </script>
